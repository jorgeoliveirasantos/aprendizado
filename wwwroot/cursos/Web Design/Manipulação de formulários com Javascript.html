Manipulação de formulários com Javascript

Neste exercício vamos adentrar um pouco no Javascript, pode ser que você tenha um pouco de dificuldade para entendê-lo, mas não se preocupe pois o próximo curso será inteiramente dedicado ao Javascript, por enquanto, apenas absorva a forma como o código é feito e o resultado que ele produz.

Podemos obter o valor de um controle por meio de scripts de maneira muito prática, devemos utilizar um seletor, como o de id getElementById(), ou o de CSS querySelector(), e podemos acessar o valor retornado por um controle com a propriedade value:

<form>
    <label for="nome">Digite seu nome:</label>
    <input type="text" id="nome"><br>
    <label for="sobrenome">Digite seu sobrenome:</label>
    <input type="text" id="sobrenome"><br>
    <input type="button" value="Cadastrar">
</form>
<script>
let nome = document.getElementById('nome');
let sobrenome = document.getElementById('sobrenome');
let btn = document.querySelector('input[type=button]');
btn.onclick = () => {
    alert("Seja bem vindo " + nome.value + " " + sobrenome.value);
}
</script>

No próximo exercício, criaremos um formulário para compra de um veículo onde selecionaremos opções de cor, marca etc. Obteremos por meio do JS os valores dos controles e os armazenaremos para exibir no resultado da compra que será exibido no início do documento e também na caixa de mensagem ao clicarmos no botão de compra.

Sem mais delongas, vamos começar com o documento HTML, note que inserimos referência ao arquivo CSS e ao arquivo Javascript no cabeçalho do documento, portanto, além do documento HTML, você deve criar também um arquivo “meuestilo.css” para criar o código de estilo e “meuscript.js” para escrever o script.

<!DOCTYPE html>
<html lang="pt-BR">
<head>
  <script src="meuscript.js"></script>
  <link rel="stylesheet" href="meuestilo.css">
  <meta charset="UTF-8">
</head>


<body>


  <h1 id="carro" onload="selecionaCarro()">


    Carro selecionado?


  </h1>


  <form>


    <fieldset>


      <label>Selecione a marca:</label>


      <br>


      <select oninput="mudaMarca(this.value)">


        <option>Ford</option>


        <option>Mercedes</option>


        <option>Chevrolet</option>


        <option>Honda</option>


        <option>Jaguar</option>


      </select>


    </fieldset>


    <fieldset>


      <label>Selecione o tipo:</label>


      <br>


      <label for="sedan">Sedan:</label>


      <input type="radio" id="sedan" 


        name="tipo"


        oninput="mudaTipo(this.id)">


      <label for="hatch">Hatch:</label>


      <input type="radio" id="hatch" 


        name="tipo"


        oninput="mudaTipo(this.id)">


      <label for="suv">SUV:</label>


      <input type="radio" id="suv" 


        name="tipo"


        oninput="mudaTipo(this.id)" checked>


    </fieldset>


    <fieldset>


      <label>Selecione a faixa de valor:</label>


      <br>


      <input type="range"


        id="valor"


        min="50000"


        max="500000"


        step="50000"


        oninput="mudaPreco(this.value)">


    </fieldset>


    <fieldset>


      <label for="">Selecione a cor:</label>


      <input type="color" oninput="mudaCor(this.value)">


    </fieldset>


    <button onclick="Comprar()">Comprar!</button>


  </form>


</body>


</html>

Com o documento HTML pronto, vamos criar o arquivo de folha de estilo, na mesma pasta crie um arquivo meuestilo.css e introduza o seguinte código:

}

}

Por fim, crie um arquivo meuscript.js e introduza o seguinte código:

let preco = "300000";
let cor = "#000000";
let marca = "Ford";
let tipo = "SUV";
function mudaPreco(val) {
    preco = val;
    selecionaCarro();
}
function mudaCor(val) {
    cor = val;
    selecionaCarro();
}
function mudaTipo(val) {
    tipo = val;
    selecionaCarro();
}
function mudaMarca(val) {
    marca = val;
    selecionaCarro();
}
function selecionaCarro() {
    document.getElementById("carro").innerHTML = `Carro selecionado: ${marca}, ${tipo}, R$: ${preco}. Cor: 
    <span style="height: 20px; width: 30px; background-color: ${cor}; display: inline-block;"></span>`;
}
function Comprar() {
    alert(`Parabens! Você comprou um lindo ${marca}, ${tipo}, por R$: ${preco}!`)
}
